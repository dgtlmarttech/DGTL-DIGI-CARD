'use client'
import React, { useState, useCallback, useEffect, useRef, useMemo } from 'react';
import { useAuthState } from 'react-firebase-hooks/auth';
import { useRouter } from 'next/navigation';
import {
  collection,
  query,
  where,
  addDoc,
  getDoc,
  doc,
  deleteDoc,
  onSnapshot,
  serverTimestamp,
  updateDoc,
} from 'firebase/firestore';
import { auth, db } from '../../../firebase/firebase'; // Assuming these are correctly exported from your firebase.js

// Toast for notifications
import { ToastContainer, toast } from 'react-toastify';
// import 'react-toastify/dist/React-toastify.css';

// Import the custom CSS file
import './Network.css';

// Import vCard utility functions
import { exportContactsToVCF } from '../../../utils/vcardUtils';

// Import the new Scanner component
import Scanner from '../../../components/Scanner'; // Assuming Scanner.jsx is in the same directory

// Simple Hamburger Icon
const HamburgerIcon = () => (
  <svg className="dc-network-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg">
    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M4 6h16M4 12h16M4 18h16"></path>
  </svg>
);

// Simple Close Icon
const CloseIcon = () => (
  <svg className="dc-network-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg">
    <path strokeLinecap="round" strokeWidth="2" d="M6 18L18 6M6 6l12 12"></path>
  </svg>
);

// Phone Icon (for direct call)
const PhoneIcon = () => (
  <svg className="dc-network-contact-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg">
    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M3 5a2 2 0 012-2h3.28a1 1 0 01.948.684l1.498 4.493a1 1 0 01-.502 1.21l-2.257 1.13a11.042 11.042 0 005.516 5.516l1.13-2.257a1 1 0 011.21-.502l4.493 1.498a1 1 0 01.684.949V19a2 2 0 01-2 2h-1C9.716 21 3 14.284 3 6V5z"></path>
  </svg>
);

// Mail Icon (for direct email)
const MailIcon = () => (
  <svg className="dc-network-contact-icon" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg">
    <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M3 8l7.89 5.26a2 2 0 002.22 0L21 8M5 19h14a2 2 0 002-2V7a2 2 0 00-2-2H5a2 2 0 00-2 2v10a2 2 0 002 2z"></path>
  </svg>
);


const NetworkPage = () => {
  const [user, loadingAuth] = useAuthState(auth);
  const router = useRouter();

  const [activeTab, setActiveTab] = useState('contacts');
  const [isMobileSidebarOpen, setIsMobileSidebarOpen] = useState(false);
  const sidebarRef = useRef(null);

  const [contacts, setContacts] = useState([]);
  const [currentContactId, setCurrentContactId] = useState(null);
  const [currentContactDetails, setCurrentContactDetails] = useState(null);

  const csvFileInputRef = useRef(null); // Ref for CSV file input

  // State for the Add/Edit Contact form
  const [contactForm, setContactForm] = useState({
    name: '',
    company: '',
    title: '',
    email: '',
    phone: '',
    notes: '',
    labels: [], // Array of label IDs
    groups: [], // Array of group IDs
    vcardString: '', // Raw vCard string from QR scan
  });
  const [isEditingContact, setIsEditingContact] = useState(false);

  // States for Labels and Groups Management
  const [labels, setLabels] = useState([]);
  const [groups, setGroups] = useState([]);
  const [newLabelName, setNewLabelName] = useState('');
  const [newLabelColor, setNewLabelColor] = useState('#000000');
  const [newGroupName, setNewGroupName] = useState('');

  // States for Search and Filter
  const [searchTerm, setSearchTerm] = useState('');
  const [sortBy, setSortBy] = useState('dateAdded'); // 'name', 'dateAdded', 'lastInteractionDate'
  const [filterLabels, setFilterLabels] = useState([]); // Array of label IDs to filter by
  const [filterGroups, setFilterGroups] = useState([]); // Array of group IDs to filter by

  // State for filter dropdown visibility
  const [isLabelsFilterOpen, setIsLabelsFilterOpen] = useState(false);
  const [isGroupsFilterOpen, setIsGroupsFilterOpen] = useState(false);

  // State for QR scanner activation (controlled by NetworkPage)
  const [isScanning, setIsScanning] = useState(false);


  const labelsFilterRef = useRef(null);
  const groupsFilterRef = useRef(null);


  // --- Firebase Authentication and Data Loading ---
  useEffect(() => {
    if (loadingAuth) return;
    if (!user) {
      toast.error('You need to be logged in to access the CRM.');
      router.push('/signin');
      return;
    }

    let unsubscribeContacts;
    let unsubscribeLabels;
    let unsubscribeGroups;

    if (user) {
      // Listen for contacts
      const contactsQuery = query(collection(db, 'contacts'), where('userId', '==', user.uid));
      unsubscribeContacts = onSnapshot(
        contactsQuery,
        (snapshot) => {
          const fetchedContacts = [];
          snapshot.forEach((doc) => {
            fetchedContacts.push({ id: doc.id, ...doc.data() });
          });
          setContacts(fetchedContacts);
        },
        (error) => {
          console.error('Error fetching contacts:', error);
          toast.error(`Error loading contacts: ${error.message}`);
        }
      );

      // Listen for labels
      const labelsQuery = query(collection(db, 'labels'), where('userId', '==', user.uid));
      unsubscribeLabels = onSnapshot(
        labelsQuery,
        (snapshot) => {
          const fetchedLabels = [];
          snapshot.forEach((doc) => {
            fetchedLabels.push({ id: doc.id, ...doc.data() });
          });
          setLabels(fetchedLabels);
        },
        (error) => {
          console.error('Error fetching labels:', error);
          toast.error(`Error loading labels: ${error.message}`);
        }
      );

      // Listen for groups
      const groupsQuery = query(collection(db, 'groups'), where('userId', '==', user.uid));
      unsubscribeGroups = onSnapshot(
        groupsQuery,
        (snapshot) => {
          const fetchedGroups = [];
          snapshot.forEach((doc) => {
            fetchedGroups.push({ id: doc.id, ...doc.data() });
          });
          setGroups(fetchedGroups);
        },
        (error) => {
          console.error('Error fetching groups:', error);
          toast.error(`Error loading groups: ${error.message}`);
        }
      );
    }
    if (activeTab !== 'scan-qr' && isScanning) {
      setIsScanning(false); // Request Scanner component to stop
    }


    return () => {
      if (unsubscribeContacts) unsubscribeContacts();
      if (unsubscribeLabels) unsubscribeLabels();
      if (unsubscribeGroups) unsubscribeGroups();
    };
  }, [user, loadingAuth, router,activeTab,isScanning]);

  // --- Tab Change Logic ---
  const handleTabChange = useCallback((tabId) => {
    setActiveTab(tabId);
    setIsMobileSidebarOpen(false); // Close sidebar on tab change
    // Close filter dropdowns on tab change
    setIsLabelsFilterOpen(false);
    setIsGroupsFilterOpen(false);

    // Stop scanner if changing away from 'scan-qr' tab
    if (tabId !== 'scan-qr' && isScanning) {
      setIsScanning(false); // Request Scanner component to stop
    }

    // Reset form and editing state when changing tabs, unless switching to add-contact for editing
    if (tabId !== 'add-contact' && tabId !== 'contact-detail') {
      setContactForm({
        name: '', company: '', title: '', email: '', phone: '', notes: '', labels: [], groups: [], vcardString: '',
      });
      setIsEditingContact(false);
      setCurrentContactId(null);
      setCurrentContactDetails(null);
    }
  }, [isScanning]);

  // --- Scanner Callbacks ---
  const handleScanSuccess = useCallback((parsedContact) => {
    // Pre-populate add contact form and switch to it
    setContactForm({
      ...parsedContact,
      labels: [], // Reset labels and groups for new contact
      groups: [],
    });
    setIsEditingContact(false); // Ensure it's treated as a new contact from scan
    handleTabChange('add-contact');
    setIsScanning(false); // Ensure scanning state is off after successful scan
  }, [handleTabChange]);

  const handleScanError = useCallback((errorMessage) => {
    console.error('Scanner Error:', errorMessage);
    // The Scanner component itself will show toasts for permission/not-found errors.
    // This callback is for additional handling if needed, or just logging.
    setIsScanning(false); // Ensure scanning state is off on error
  }, []);

  const handleScannerStopped = useCallback(() => {
    setIsScanning(false); // Ensure local state is in sync
    // No toast here, as scanner component already toasts "stopped"
  }, []);


  // Handle clicks outside sidebar and filter dropdowns to close them on mobile
  useEffect(() => {
    const handleClickOutside = (event) => {
      // Close mobile sidebar
      if (sidebarRef.current && !sidebarRef.current.contains(event.target)) {
        if (!event.target.closest('.dc-network-mobile-menu-button')) {
          setIsMobileSidebarOpen(false);
        }
      }

      // Close labels filter dropdown
      if (labelsFilterRef.current && !labelsFilterRef.current.contains(event.target)) {
        setIsLabelsFilterOpen(false);
      }

      // Close groups filter dropdown
      if (groupsFilterRef.current && !groupsFilterRef.current.contains(event.target)) {
        setIsGroupsFilterOpen(false);
      }
    };

    document.addEventListener('mousedown', handleClickOutside);

    // Manage body overflow for mobile sidebar
    if (isMobileSidebarOpen) {
      document.body.classList.add('dc-network-overflow-hidden');
    } else {
      document.body.classList.remove('dc-network-overflow-hidden');
    }

    return () => {
      document.removeEventListener('mousedown', handleClickOutside);
      document.body.classList.remove('dc-network-overflow-hidden');
    };
  }, [isMobileSidebarOpen, isLabelsFilterOpen, isGroupsFilterOpen]);


  // --- Add/Edit Contact Form Handlers ---
  const handleContactFormChange = (e) => {
    const { name, value } = e.target;
    setContactForm((prev) => ({ ...prev, [name]: value }));
  };

  const handleLabelSelect = (labelId) => {
    setContactForm((prev) => {
      const currentLabels = prev.labels || [];
      if (currentLabels.includes(labelId)) {
        return { ...prev, labels: currentLabels.filter((id) => id !== labelId) };
      } else {
        return { ...prev, labels: [...currentLabels, labelId] };
      }
    });
  };

  const handleGroupSelect = (groupId) => {
    setContactForm((prev) => {
      const currentGroups = prev.groups || [];
      if (currentGroups.includes(groupId)) {
        return { ...prev, groups: currentGroups.filter((id) => id !== groupId) };
      } else {
        return { ...prev, groups: [...currentGroups, groupId] };
      }
    });
  };

  const handleAddOrUpdateContactSubmit = async (e) => {
    e.preventDefault();
    if (!user) {
      toast.error('Authentication required to save contacts.');
      return;
    }
    if (!contactForm.name.trim()) {
      toast.error('Contact name is required!');
      return;
    }

    const contactData = {
      name: contactForm.name.trim(),
      company: contactForm.company.trim(),
      title: contactForm.title.trim(),
      email: contactForm.email.trim(),
      phone: contactForm.phone.trim(),
      notes: contactForm.notes.trim(),
      labels: contactForm.labels,
      groups: contactForm.groups,
      userId: user.uid,
      lastInteractionDate: serverTimestamp(),
      ...(contactForm.vcardString && { vcardString: contactForm.vcardString }),
    };

    try {
      if (isEditingContact && currentContactId) {
        await updateDoc(doc(db, 'contacts', currentContactId), contactData);
        toast.success('Contact updated successfully! 🎉');
      } else {
        await addDoc(collection(db, 'contacts'), {
          ...contactData,
          source: contactForm.vcardString ? 'Scanned QR' : 'Manual Add',
          dateAdded: serverTimestamp(),
        });
        toast.success('Contact added successfully! 🎉');
      }
      setContactForm({ name: '', company: '', title: '', email: '', phone: '', notes: '', labels: [], groups: [], vcardString: '' });
      setIsEditingContact(false);
      setCurrentContactId(null);
      setCurrentContactDetails(null);
      handleTabChange('contacts');
    } catch (error) {
      console.error('Error saving contact:', error);
      toast.error(`Error saving contact: ${error.message}`);
    }
  };

  // --- Contact Detail View Handlers ---
  const displayContactDetails = useCallback(async (contactId) => {
    if (!user) {
      toast.error('Authentication required to view contact details.');
      return;
    }
    setCurrentContactId(contactId);
    handleTabChange('contact-detail');

    try {
      const contactDocRef = doc(db, 'contacts', contactId);
      const contactSnap = await getDoc(contactDocRef);

      if (contactSnap.exists() && contactSnap.data().userId === user.uid) {
        const data = contactSnap.data();
        setCurrentContactDetails({ id: contactSnap.id, ...data });
      } else {
        toast.error('Contact not found or unauthorized.');
        handleTabChange('contacts');
      }
    } catch (error) {
      console.error('Error fetching contact details:', error);
      toast.error(`Error fetching details: ${error.message}`);
      handleTabChange('contacts');
    }
  }, [user, handleTabChange]);

  const handleDeleteContact = async () => {
    if (!currentContactId || !user) {
      toast.error('No contact selected for deletion or not authenticated.');
      return;
    }

    toast.warn(
      <div>
        <p>Are you sure you want to delete this contact?</p>
        <button
          className="dc-network-button dc-network-button-danger"
          onClick={async () => {
            try {
              await deleteDoc(doc(db, 'contacts', currentContactId));
              toast.success('Contact deleted successfully!');
              setCurrentContactId(null);
              setCurrentContactDetails(null);
              handleTabChange('contacts');
              toast.dismiss();
            } catch (error) {
              console.error('Error deleting contact:', error);
              toast.error(`Error deleting contact: ${error.message}`);
            }
          }}
        >
          Yes, Delete
        </button>
        <button
          className="dc-network-button dc-network-button-secondary"
          onClick={() => toast.dismiss()}
        >
          Cancel
        </button>
      </div>,
      {
        autoClose: false,
        closeButton: false,
        draggable: false,
      }
    );
  };

  const handleEditContact = () => {
    if (currentContactDetails) {
      setContactForm({
        name: currentContactDetails.name || '',
        company: currentContactDetails.company || '',
        title: currentContactDetails.title || '',
        email: currentContactDetails.email || '',
        phone: currentContactDetails.phone || '',
        notes: currentContactDetails.notes || '',
        labels: currentContactDetails.labels || [],
        groups: currentContactDetails.groups || [],
        vcardString: currentContactDetails.vcardString || '',
      });
      setIsEditingContact(true);
      handleTabChange('add-contact');
    } else {
      toast.error('No contact selected to edit.');
    }
  };


  // --- CSV Import Logic ---
  const handleImportCsv = async () => {
    if (!user) {
      toast.error('Authentication required to import contacts.');
      return;
    }

    const file = csvFileInputRef.current?.files[0];
    if (!file) {
      toast.error('Please select a CSV file to import.');
      return;
    }

    if (file.type !== 'text/csv') {
      toast.error('Please upload a valid CSV file.');
      return;
    }

    const reader = new FileReader();
    reader.onload = async (e) => {
      const text = e.target.result;
      const lines = text.split('\n').filter(line => line.trim() !== '');

      if (lines.length === 0) {
        toast.error('CSV file is empty.');
        return;
      }

      // Assuming header row: Name,Email,Phone,Company,Title,Notes
      const headers = lines[0].split(',').map(h => h.trim().toLowerCase());
      const expectedHeaders = ['name', 'email', 'phone', 'company', 'title', 'notes'];

      // Basic header validation
      const missingHeaders = expectedHeaders.filter(eh => !headers.includes(eh));
      if (missingHeaders.length > 0) {
        toast.error(`Missing required CSV headers: ${missingHeaders.join(', ')}. Expected: ${expectedHeaders.join(', ')}`);
        return;
      }

      const contactsToImport = [];
      for (let i = 1; i < lines.length; i++) {
        const values = lines[i].split(',');
        if (values.length !== headers.length) {
          console.warn(`Skipping malformed row ${i + 1}: ${lines[i]}`);
          toast.warn(`Skipped row ${i + 1} due to incorrect column count.`);
          continue;
        }

        const contact = {};
        headers.forEach((header, index) => {
          contact[header] = values[index] ? values[index].trim() : '';
        });

        if (!contact.name) {
          console.warn(`Skipping row ${i + 1}: Name is empty.`);
          toast.warn(`Skipped row ${i + 1} as contact name is empty.`);
          continue;
        }

        contactsToImport.push({
          name: contact.name,
          email: contact.email,
          phone: contact.phone,
          company: contact.company,
          title: contact.title,
          notes: contact.notes,
          labels: [], // CSV import doesn't support labels/groups directly in this basic version
          groups: [],
          userId: user.uid,
          source: 'CSV Import',
          dateAdded: serverTimestamp(),
          lastInteractionDate: serverTimestamp(),
        });
      }

      if (contactsToImport.length === 0) {
        toast.info('No valid contacts found in the CSV file to import.');
        return;
      }

      let importedCount = 0;
      for (const contact of contactsToImport) {
        try {
          await addDoc(collection(db, 'contacts'), contact);
          importedCount++;
        } catch (error) {
          console.error('Error adding contact from CSV:', contact, error);
          toast.error(`Failed to import contact ${contact.name}: ${error.message}`);
        }
      }

      toast.success(`Successfully imported ${importedCount} contacts!`);
      csvFileInputRef.current.value = ''; // Clear file input
      handleTabChange('contacts'); // Go back to contacts list
    };

    reader.onerror = () => {
      toast.error('Error reading CSV file.');
    };

    reader.readAsText(file);
  };


  // --- Labels Management ---
  const handleAddLabel = async (e) => {
    e.preventDefault();
    if (!user) {
      toast.error('Authentication required to add labels.');
      return;
    }
    if (!newLabelName.trim()) {
      toast.error('Label name cannot be empty.');
      return;
    }

    try {
      await addDoc(collection(db, 'labels'), {
        labelName: newLabelName.trim(),
        color: newLabelColor,
        userId: user.uid,
        createdAt: serverTimestamp(),
      });
      toast.success('Label added successfully!');
      setNewLabelName('');
      setNewLabelColor('#000000');
    } catch (error) {
      console.error('Error adding label:', error);
      toast.error(`Error adding label: ${error.message}`);
    }
  };

  const handleDeleteLabel = async (labelId) => {
    if (!user) {
      toast.error('Authentication required to delete labels.');
      return;
    }
    toast.warn(
      <div>
        <p>Are you sure you want to delete this label? It will be removed from all contacts.</p>
        <button
          className="dc-network-button dc-network-button-danger"
          onClick={async () => {
            try {
              // Remove this label from all contacts that use it
              const contactsToUpdate = contacts.filter(contact => contact.labels && contact.labels.includes(labelId));
              const batchUpdates = contactsToUpdate.map(contact => {
                const updatedLabels = contact.labels.filter(l => l !== labelId);
                return updateDoc(doc(db, 'contacts', contact.id), { labels: updatedLabels });
              });
              await Promise.all(batchUpdates);

              await deleteDoc(doc(db, 'labels', labelId));
              toast.success('Label deleted successfully!');
              toast.dismiss();
            } catch (error) {
              console.error('Error deleting label:', error);
              toast.error(`Error deleting label: ${error.message}`);
            }
          }}
        >
          Yes, Delete
        </button>
        <button className="dc-network-button dc-network-button-secondary" onClick={() => toast.dismiss()}>Cancel</button>
      </div>,
      { autoClose: false, closeButton: false, draggable: false }
    );
  };

  // --- Groups Management ---
  const handleAddGroup = async (e) => {
    e.preventDefault();
    if (!user) {
      toast.error('Authentication required to add groups.');
      return;
    }
    if (!newGroupName.trim()) {
      toast.error('Group name cannot be empty.');
      return;
    }

    try {
      await addDoc(collection(db, 'groups'), {
        groupName: newGroupName.trim(),
        userId: user.uid,
        createdAt: serverTimestamp(),
      });
      toast.success('Group added successfully!');
      setNewGroupName('');
    } catch (error) {
      console.error('Error adding group:', error);
      toast.error(`Error adding group: ${error.message}`);
    }
  };

  const handleDeleteGroup = async (groupId) => {
    if (!user) {
      toast.error('Authentication required to delete groups.');
      return;
    }
    toast.warn(
      <div>
        <p>Are you sure you want to delete this group? It will be removed from all contacts.</p>
        <button
          className="dc-network-button dc-network-button-danger"
          onClick={async () => {
            try {
              // Remove this group from all contacts that use it
              const contactsToUpdate = contacts.filter(contact => contact.groups && contact.groups.includes(groupId));
              const batchUpdates = contactsToUpdate.map(contact => {
                const updatedGroups = contact.groups.filter(g => g !== groupId);
                return updateDoc(doc(db, 'contacts', contact.id), { groups: updatedGroups });
              });
              await Promise.all(batchUpdates);

              await deleteDoc(doc(db, 'groups', groupId));
              toast.success('Group deleted successfully!');
              toast.dismiss();
            } catch (error) {
              console.error('Error deleting group:', error);
              toast.error(`Error deleting group: ${error.message}`);
            }
          }}
        >
          Yes, Delete
        </button>
        <button className="dc-network-button dc-network-button-secondary" onClick={() => toast.dismiss()}>Cancel</button>
      </div>,
      { autoClose: false, closeButton: false, draggable: false }
    );
  };

  // --- Search and Filter Logic ---
  const filteredAndSortedContacts = useMemo(() => {
    let filtered = contacts;

    // Apply search term across name, email, phone
    if (searchTerm.trim()) {
      const lowerCaseSearchTerm = searchTerm.toLowerCase();
      filtered = filtered.filter(contact =>
        (contact.name && contact.name.toLowerCase().includes(lowerCaseSearchTerm)) ||
        (contact.email && contact.email.toLowerCase().includes(lowerCaseSearchTerm)) ||
        (contact.phone && contact.phone.toLowerCase().includes(lowerCaseSearchTerm))
      );
    }

    // Apply label filters
    if (filterLabels.length > 0) {
      filtered = filtered.filter(contact =>
        contact.labels && filterLabels.every(filterLabelId => contact.labels.includes(filterLabelId))
      );
    }

    // Apply group filters
    if (filterGroups.length > 0) {
      filtered = filtered.filter(contact =>
        contact.groups && filterGroups.every(filterGroupId => contact.groups.includes(filterGroupId))
      );
    }

    // Apply sorting
    filtered.sort((a, b) => {
      if (sortBy === 'name') {
        return (a.name || '').localeCompare(b.name || '');
      } else if (sortBy === 'dateAdded') {
        const dateA = a.dateAdded?.toDate ? a.dateAdded.toDate().getTime() : 0;
        const dateB = b.dateAdded?.toDate ? b.dateAdded.toDate().getTime() : 0;
        return dateB - dateA; // Newest first
      } else if (sortBy === 'lastInteractionDate') {
        const dateA = a.lastInteractionDate?.toDate ? a.lastInteractionDate.toDate().getTime() : 0;
        const dateB = b.lastInteractionDate?.toDate ? b.lastInteractionDate.toDate().getTime() : 0;
        return dateB - dateA; // Newest first
      }
      return 0;
    });

    return filtered;
  }, [contacts, searchTerm, sortBy, filterLabels, filterGroups]);

  // --- Analytics Calculation ---
  const contactAnalytics = useMemo(() => {
    const totalContacts = contacts.length;

    const contactsBySource = contacts.reduce((acc, contact) => {
      const source = contact.source || 'Unknown';
      acc[source] = (acc[source] || 0) + 1;
      return acc;
    }, {});

    const contactsByLabel = labels.reduce((acc, label) => {
      const count = contacts.filter(contact => contact.labels && contact.labels.includes(label.id)).length;
      acc[label.labelName] = { count, color: label.color };
      return acc;
    }, {});

    const contactsByGroup = groups.reduce((acc, group) => {
      const count = contacts.filter(contact => contact.groups && contact.groups.includes(group.id)).length;
      acc[group.groupName] = { count };
      return acc;
    }, {});

    return {
      totalContacts,
      contactsBySource,
      contactsByLabel,
      contactsByGroup,
    };
  }, [contacts, labels, groups]);

  // --- VCF Export Handlers ---
  const handleExportSingleVCF = () => {
    if (currentContactDetails) {
      exportContactsToVCF([currentContactDetails], `${currentContactDetails.name || 'contact'}.vcf`);
      toast.success('Contact exported to VCF!');
    } else {
      toast.error('No contact selected for export.');
    }
  };

  const handleExportAllVCF = () => {
    if (filteredAndSortedContacts.length > 0) {

                  Back to Contacts
                </button>
                <button
                    className="dc-network-button dc-network-button-info"
                    onClick={handleExportSingleVCF}
                >
                    Export VCF
                </button>
              </div>
            </section>
          )}
        </main>
      </div>
    </div>
  );
};

export default NetworkPage;
